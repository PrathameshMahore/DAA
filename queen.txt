// N-QUEENS
#include <stdio.h>
#include <math.h>

int x[4]; // solution stored array

int place(int k, int i);
void print_solution(int x[], int n);

void nQueens(int k, int n) {
    for (int i = 1; i <= n; i++) { // column
        if (place(k, i)) {
            x[k] = i;
            if (k == n) {
                print_solution(x, n);
            } else {
                nQueens(k + 1, n);
            }
        }
    }
}

int place(int k, int i) {
    for (int j = 1; j <= k - 1; j++) {
        // Check if the current placement conflicts with previous placements
        if (x[j] == i || abs(x[j] - i) == abs(j - k)) {
            return 0;
        }
    }
    return 1;
}

void print_solution(int x[], int n) {
    printf("Solution: -->\n");
    for (int i = 1; i <= n; i++) {
        printf("%d\t", x[i]);
    }
    printf("\n");
}

int main() {
    int n;
    printf("Enter the size of the chess board:\n");
    scanf("%d", &n);
    
    int k = 1;
    nQueens(k, n);
    
    return 0;
}
